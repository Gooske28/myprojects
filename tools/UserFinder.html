Beschrijving:

Een programma dat op Facebook, Twitter, Instagram en LinkeIn naar een gebruikersnaam zoekt die je invoert. Je hebt enkel requests nodig.

Code:

import requests
from concurrent.futures import ThreadPoolExecutor

def check_username(username):
    urls = [
        f"https://www.facebook.com/{username}",
        f"https://twitter.com/{username}",
        f"https://www.instagram.com/{username}",
        f"https://www.linkedin.com/in/{username}",
    ]
    for url in urls:
        response = requests.get(url)
        if response.status_code == 200:
            print(f"Gebruikersnaam gevonden: {url}")
        else:
            print(f"Geen gebruikersnaam gevonden op: {url}")

if __name__ == "__main__":
    username = input("Voer de gebruikersnaam in om te controleren: ")  # Hier kan de gebruiker een gebruikersnaam invoeren
    with ThreadPoolExecutor() as executor:
        future = executor.submit(check_username, username)
        future.result()  # Wacht tot de thread klaar is

    input("\nDruk op Enter om af te sluiten...")  # Voorkomt automatisch afsluiten
