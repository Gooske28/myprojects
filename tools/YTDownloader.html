Beschrijving:

Een programma dat liedjes van YouTube kan downloaden. Je moet de downloadlocatie in de source code instellen en dan moet je gewoon de link in de terminal invoeren. Je hebt pytubefix en os nodig.

Code:

from pytubefix import YouTube  # Gebruik pytubefix i.p.v. pytube
import os

def download_mp3(output_path):
    try:
        video_url = input("Voer de YouTube video URL in: ").split("&pp=")[0]  # Verwijder extra parameters
        yt = YouTube(video_url)
        audio_stream = yt.streams.filter(only_audio=True).first()
        
        # Gebruik raw string of forward slashes voor pad
        output_file = audio_stream.download(output_path=output_path)
        
        # Hernoem naar .mp3
        base, _ = os.path.splitext(output_file)
        new_file = base + '.mp3'
        os.rename(output_file, new_file)
        print(f"Gedownload: {new_file}")
        
    except Exception as e:
        print(f"Fout: {str(e)}")

# Pad aanpassen naar raw string of forward slashes
output_path = r"D:\Muziek"  # Raw string voor Windows-paden
os.makedirs(output_path, exist_ok=True)
download_mp3(output_path)
